%-=-=-= EXAMPLE
\begin{example}[id:20151012-201647] \label{20151012-201647}\index{Example!20151012-201647} \hfill \\
Find $f(2)$ given $f(x)=2x^3-3x^2-12x+1$
\soln

\solnsteps
\begin{align*}
f(x) & =2x^3 + \neg 3x^2 + \neg 12x+1 && \text{DOS} \eqref{eq:dos1} \\
f(2) &= 2\farg{2}^3 + \neg 3\farg{2}^2 + \neg 12 \farg{2} +1 && \text{SPE} \eqref{eq:spe} \\
f(2) &= 2(8) + \neg 3(4) + \neg 12(2) + 1 && \text{OOE} \eqref{eq:ooe} \\
f(2) &= 16 + \neg 12 + \neg 24 +1 && \text{OOM} \eqref{eq:oom} \\
f(2) &= -19 && \text{OOA} \eqref{eq:ooa} \\
\end{align*}

\qdepend

\begin{example}[id:20151012-203549] \label{20151012-203549}\index{Example!20151012-203549} \hfill \\

\soln

\solnsteps

f(\neg 1) &= \neg 2 + \neg 3 + 12 +1 && \text{OOM} \eqref{eq:oom} \\
f(\neg 1) &= 8 && \text{OOA} \eqref{eq:ooa} \\
\end{align*}

\begin{property}[Cosine Inverse (ArcCos)]\index{Cosine Inverse}
\begin{subequations}
\begin{align}
\cos\inv \left(\cos \theta \right) = \theta \label{eq:arccos}
\end{align}
\end{subequations}
\end{property}

%-=-=-= DEFINITION
\begin{property}[Sine Inverse (ArcSin)]\index{Sine Inverse}

\sin\inv \left(\sin \theta \right) = \theta  \label{eq:arcsin}
\end{align}
\end{subequations}
\end{property}

%-=-=-= DEFINITION
